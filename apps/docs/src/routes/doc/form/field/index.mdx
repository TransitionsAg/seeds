# Field

Compound field component returned by `useFormComponents`.

## Usage

```tsx
function LoginForm() {
  const form = useForm({
    initialValues: { email: "" },
    onSubmit: (value) => console.log(value),
  });
  const { Form, Field } = useFormComponents(form);

  return (
    <Form>
      <Field name="email">
        <Field.Label>Email</Field.Label>
        <Field.Input type="email" />
        <Field.Error />
      </Field>
    </Form>
  );
}
```
