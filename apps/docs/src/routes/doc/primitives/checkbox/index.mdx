# Checkbox

An unstyled checkbox primitive wrapping <code>@zag-js/checkbox</code>.

## Installation

```bash
pnpm add @transitionsag/primitives/checkbox
```

## Usage

```tsx
import { Checkbox } from "@transitionsag/primitives/checkbox";

<Checkbox id="terms">
  <Checkbox.Control>
    <Checkbox.Indicator>✓</Checkbox.Indicator>
  </Checkbox.Control>
  <Checkbox.Label>Accept terms</Checkbox.Label>
  <Checkbox.HiddenInput />
</Checkbox>;
```
